    if number < 2:
        return False
    i = 2
    prime = True

def get_eratos(list_max):
    o_list = list(range(2,list_max+1))
    p_list = []
    while o_list[0]*o_list[0] <= list_max :
        p_list.append(o_list[0])
        o_list = list(filter(lambda x : x % o_list[0] != 0, o_list))
    p_list += o_list
    return p_list
