def mat_m(m_1,m_2):
    res_m = [m_1[0]*m_2[0]+m_1[1]*m_2[2],
             m_1[0]*m_2[1]+m_1[1]*m_2[3],
             m_1[2]*m_2[0]+m_1[3]*m_2[2],
             m_1[2]*m_2[1]+m_1[3]*m_2[3]]
    return res_m

def mat_pow(m,n):
    if n == 1:
        return m
    k = n // 2
    return mat_m(mat_pow(m,k),mat_pow(m,n-k))

def fib(n):
    mat_f = [1,1,1,0]
    return mat_pow(mat_f,n)[0]
